#+begin_src Haskell
import qualified Data.Map as M

[ ("M-u", toggleFullscreen) ]

toggleFullscreen :: X ()
toggleFullscreen =
  withWindowSet $ \ws ->
    withFocused $ \w -> do
      let fullRect = W.RationalRect 0 0 1 1
      let isFullFloat = w `M.lookup` W.floating ws == Just fullRect
      windows $ if isFullFloat then W.sink w else W.float w fullRect

#+end_src
